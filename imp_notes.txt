SELECT customers.first_name, customers.last_name, company_orders.date, company_orders.total_amount 
FROM customers INNER JOIN company_orders 
ON customers.id = company_orders.customer_id;

SELECT customers.first_name, customers.last_name, company_orders.date, company_orders.total_amount 
FROM customers LEFT JOIN company_orders 
ON customers.id = company_orders.customer_id;
